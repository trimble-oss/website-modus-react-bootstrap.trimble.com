<CodeBlock title="Bordered Table" code={TableBordered}>

</CodeBlock>

<CodeBlock title="Borderless Table" code={TableBorderless}>
  Add prop <code>borderless </code> to remove borders to your table.
</CodeBlock>

<CodeBlock title="Hoverable Rows" code={TableHoverable}>
